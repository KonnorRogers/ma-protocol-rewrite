import EMTLevel from '../../components/emtLevel/emtLevel.js'

<EMTLevel level="basic">
  <ul>
    <li>1.0 Routine Patient Care</li>
    <li>
      Provide rapid cooling as soon as possible.
    </li>
    <li>
      <strong>CAUTION:</strong> Do not over-chill patient, observe for shivering. If shivering occurs,
      discontinue active cooling procedures.
    </li>
    <li>Move patient to cool area.</li>
    <li>Loosen or remove all unnecessary clothing, while protecting privacy.</li>
    <li>Apply cool packs to armpits, neck and groin.</li>
    <li>Use evaporation techniques if possible (fans, open windows).</li>
    <li>Keep skin wet by applying water with wet towels or sponges.</li>
    <li>
      For Heat Cramps and/or Heat Exhaustion: administer water or oral re-hydrationelectrolyte
      solution if patient is alert and has a normal gag reflex and can
      swallow easily. Elevate legs of supine patient with heat exhaustion.
    </li>
  </ul>
</EMTLevel>

<EMTLevel level="advanced">
  <ul>
    <li>Consider 500mL fluid bolus for dehydration even if vital signs are normal.</li>
    <li><strong>Pediatrics:</strong> 20mL/kg bolus, if indicated.</li>
  </ul>
</EMTLevel>

<EMTLevel level="paramedic">
  <ul>
    <li>Follow above protocols</li>
  </ul>
</EMTLevel>
